local Players = game:GetService("Players")

local function onToolEquipped(tool, player)
    if tool:IsA("Tool") and tool:FindFirstChild("Handle") then
        local handle = tool.Handle
        -- Move only other players' tool handles 50 studs underground
        handle.Position = handle.Position - Vector3.new(0, 50, 0)
    end
end

local function onCharacterAdded(character)
    local player = Players:GetPlayerFromCharacter(character)
    if player and player ~= game:GetService("RunService"):GetRunMode() then
        character.ChildAdded:Connect(function(child)
            if child:IsA("Tool") then
                onToolEquipped(child, player)
            end
        end)
    end
end

Players.PlayerAdded:Connect(function(player)
    player.CharacterAdded:Connect(onCharacterAdded)
end)
